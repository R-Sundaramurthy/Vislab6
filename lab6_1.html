#________________________________________ Permutation by HC____________________________________________

# adjacency representation

netm <- get.adjacency(net, attr="weight", sparse=F)

# colnames
colnames(netm) <- V(net)$label

# calculate distances
rowdist<-dist(netm)

# get order

#set seed
